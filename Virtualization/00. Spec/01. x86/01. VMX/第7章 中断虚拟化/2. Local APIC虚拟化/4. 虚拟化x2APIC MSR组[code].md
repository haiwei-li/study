	switch (kvm_get_apic_mode(vcpu)) {

		}

		if (cpu_has_vmx_virtualize_x2apic_mode())

				SECONDARY_EXEC_VIRTUALIZE_X2APIC_MODE;

"`virtualize APIC accesses`" 与 "`virtualize x2APIC mode`" 位不能同时设为 1 值.
